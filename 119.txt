class Solution {
public:
    Node* connect(Node* root) {
        Node* leftmost = root;

        while (leftmost != nullptr) {
            Node dummy(0);
            Node* tail = &dummy;
            Node* curr = leftmost;

            while (curr != nullptr) {
                if (curr->left) {
                    tail->next = curr->left;
                    tail = tail->next;
                }
                if (curr->right) {
                    tail->next = curr->right;
                    tail = tail->next;
                }
                curr = curr->next;
            }

            leftmost = dummy.next;
        }

        return root;   // <-- make sure this line is present, right before the closing brace
    }
};
