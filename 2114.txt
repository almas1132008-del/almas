class Solution {
public:
    int mostWordsFound(vector<string>& sentences) {
        int maxWords = 0;

        for (string& sentence : sentences) {
            int wordCount = 1; // start at 1 since n words = n-1 spaces
            for (char c : sentence) {
                if (c == ' ') {
                    wordCount++;
                }
            }
            maxWords = max(maxWords, wordCount);
        }

        return maxWords;
    }
};
