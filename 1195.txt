#include <semaphore.h>
#include <functional>
using namespace std;

class FizzBuzz {
private:
    int n;
    sem_t semFizz, semBuzz, semFizzBuzz, semNumber;

public:
    FizzBuzz(int n) {
        this->n = n;
        sem_init(&semFizz, 0, 0);
        sem_init(&semBuzz, 0, 0);
        sem_init(&semFizzBuzz, 0, 0);
        sem_init(&semNumber, 0, 1);
    }

    void fizz(function<void()> printFizz) {
        for (int i = 1; i <= n; i++) {
            if (i % 3 == 0 && i % 5 != 0) {
                sem_wait(&semFizz);
                printFizz();
                sem_post(&semNumber);
            }
        }
    }

    void buzz(function<void()> printBuzz) {
        for (int i = 1; i <= n; i++) {
            if (i % 5 == 0 && i % 3 != 0) {
                sem_wait(&semBuzz);
                printBuzz();
                sem_post(&semNumber);
            }
        }
    }

    void fizzbuzz(function<void()> printFizzBuzz) {
        for (int i = 1; i <= n; i++) {
            if (i % 15 == 0) {
                sem_wait(&semFizzBuzz);
                printFizzBuzz();
                sem_post(&semNumber);
            }
        }
    }

    void number(function<void(int)> printNumber) {
        for (int i = 1; i <= n; i++) {
            sem_wait(&semNumber);
            if (i % 15 == 0) {
                sem_post(&semFizzBuzz);
            } else if (i % 3 == 0) {
                sem_post(&semFizz);
            } else if (i % 5 == 0) {
                sem_post(&semBuzz);
            } else {
                printNumber(i);
                sem_post(&semNumber);
            }
        }
    }
};
