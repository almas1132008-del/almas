class Solution {
public:
    Node* connect(Node* root) {
        Node* leftmost = root;
        
        while (leftmost != nullptr) {
            Node* curr = leftmost;
            Node* dummy = new Node(); // dummy head for next level
            Node* tail = dummy;
            
            while (curr != nullptr) {
                if (curr->left) {
                    tail->next = curr->left;
                    tail = tail->next;
                }
                if (curr->right) {
                    tail->next = curr->right;
                    tail = tail->next;
                }
                curr = curr->next;
            }
            
            leftmost = dummy->next;
            delete dummy;
        }
        
        return root;
    }
};
